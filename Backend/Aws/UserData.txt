#!/bin/bash
set -x
yum update -y

# Install Docker
dnf install docker -y
systemctl start docker
systemctl enable docker
usermod -a -G docker ec2-user

# Install Docker Compose
curl -L "https://github.com/docker/compose/releases/download/1.29.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
chmod +x /usr/local/bin/docker-compose

# Navigate to the application directory
cd 

# Check if necessary files exist
if [ -f Dockerfile ] && [ -f docker-compose.yaml ] && [ -f dietideals-server.jar ]; then
    # Run Docker Compose
    docker-compose up --build -d
else
    echo "Necessary files for Docker Compose are missing. Please ensure Dockerfile, docker-compose.yaml, and dietideals-server.jar are present."
fi
